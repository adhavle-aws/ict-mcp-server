Create Apex Action

Create an Apex class for Agentforce:

public class AWSArchitectAction {
    @InvocableMethod(label='Generate AWS Infrastructure')
    public static List<Result> generate(List<Request> requests) {
        Request req = requests[0];
        
        Map<String, Object> payload = new Map<String, Object>{
            'id' => String.valueOf(DateTime.now().getTime()),
            'tool' => 'build_cfn_template',
            'arguments' => new Map<String, Object>{
                'prompt' => req.requirements,
                'format' => 'yaml'
            }
        };
        
        Result result = new Result();
        result.template = callWebSocket(payload);
        return new List<Result>{ result };
    }
    
    public class Request {
        @InvocableVariable(required=true)
        public String requirements;
    }
    
    public class Result {
        @InvocableVariable
        public String template;
    }
}
